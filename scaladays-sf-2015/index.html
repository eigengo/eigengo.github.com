Exercise in exercise analysis

 

Jan Macháček @honzam399

Looking at our kick-ass system from the outside, it is still comfortable. We have mobile and web clients accessing some API—typically a REST API. So, what's the problem? Why do we need to even think about distributed systems?

First off, we need to be able to deal with a lot of data being sent to us. The days of patient users are gone. (Well, they were never here in the first place.)

The devices that our users have can send significant stream of requests to our system. Some of the devices are not that clever (think internet-of-things devices), so they won't back off when our service gets into trouble.

Some other application simply require processing all the data immediately as it comes in; any delays or misses make the system useless.

To handle the load while remaining responsive, resilient and elastic, we must abandon the request-response world. By request-response, I mean typical CRUD operations you might have done yourselves too. A world where I send a request and receive the result in the response.

Instead, we need to move to the world where the system pushes events to us. Some events may arrive as a result of the applications making requests.

val accelData = new CodecPrimitive[E] {
  val ignore1 = IgnoreCodecPrimitive(1)
  val int13 = IntCodecPrimitive(13, signed = true, 
                                ByteOrdering.BigEndian)
  override val bits = 40
  override def decode(buffer: BitVector): 
    String \/ (BitVector, E) = for {
      (b1, _) ← ignore1.decode(buffer)
      (b2, z) ← int13.decode(b1)
      (b3, y) ← int13.decode(b2)
      (b4, x) ← int13.decode(b3)
    } yield (b4, (x, y, z))
}
          

trait SensorDataDecoder[+A] {
  def supports(b: BitVector): Boolean
  def decode(b: BitVector): String \/ (BitVector, A)
}
          

trait RootSensorDataDecoder[A] 
  extends SensorDataDecoder[A] {

  final def decodeAll(b: BitVector): String \/ A = {
    decode(bits).flatMap {
      case (BitVector.empty, v) ⇒ \/.right(v)
      case (_, _) ⇒ \/.left("Undecoded bits")
    }
  }

}

def model: FlowGraph = ... else FlowGraph { implicit builder ⇒
  val split = Unzip[SensorNetValue, ActorRef]
  val listener = Broadcast[ActorRef]
  val event = Broadcast[List[BindToSensors]]
  Source(ActorPublisher(self)) ~> split.in
  split.left ~> 
    workflow.transform(() ⇒ SlidingWindow[BindToSensors](2)) ~>
    event
  split.right ~> listener
  for (query ← toWatch) {
    val join = Zip[Option[ClassifiedExercise], ActorRef]
    event ~> evaluate(query) ~> join.left
    listener ~> join.right
    join.out ~> Flow[(Option[ClassifiedExercise], ActorRef)]
      .filter(_._1.nonEmpty)
      .map { case (ex, ref) ⇒ (ex.get, ref) } ~> 
        Sink.foreach[(ClassifiedExercise, ActorRef)] { 
          case (exercise, ref) => ref ! exercise 
        }
  }
}
          

def evaluate(query: Query): Flow[List[BindToSensors], QueryValue] = {
  var currentState: Query = query
  var stableState: Option[QueryValue] = None
  Flow[List[BindToSensors]].mapAsync { sensorData ⇒
    sensorData match {
      case _ if stableState.isDefined => stableState.get
      case List(event) ⇒
        evaluateQuery(currentState)(event, lastState = true)
      case List(event, _) ⇒
        evaluateQuery(currentState)(event, lastState = false)
    } match {
      case UnstableValue(nextQuery) ⇒ async {
        val validQuery = prover.valid(nextQuery)
        val satisfiableQuery = prover.satisfiable(nextQuery)
        val simplifiedQuery = prover.simplify(nextQuery)

        if (await(validQuery)) {
          StableValue(result = true)
        } else if (await(satisfiableQuery)) {
          currentState = await(simplifiedQuery)
          UnstableValue(currentState)
        } else {
          StableValue(result = false)
        }
      }
      case value: StableValue ⇒
        stableState = Some(value)
        Future(value)
    }
  }
}

val workflow = {
  val in = UndefinedSource[SensorNetValue]
  val out = UndefinedSink[BindToSensors]

  PartialFlowGraph { implicit builder ⇒
    val classifier = Tap.identifyEvent
    val split = Broadcast[SensorNetValue]
    val merge = Zip[Set[Fact], SensorNetValue]

    in ~> split
    split ~> Flow[SensorNetValue]
      .mapConcat(_.toMap(tapSensor)
                   .find(_.isInstanceOf[AccelerometerValue])
                   .asInstanceOf[Option[AccelerometerValue]].toList)
      .via(classifier.map(_.toSet)) ~> merge.left
    split ~> merge.right
    merge.out ~> Flow[(Set[Fact], SensorNetValue)].map { 
      case (facts, data) ⇒ BindToSensors(facts, ..., data) 
    } ~> out
  }.toFlow(in, out)
}
          

class GestureWorkflows(name: String, config: Config) 
  extends SVMClassifier {
  
  def identifyEvent: Flow[AccelerometerValue, Option[Fact]] =
    Flow[AccelerometerValue]
      .transform(() ⇒ SlidingWindow[AccelerometerValue](windowSize))
      .map { (sample: List[AccelerometerValue]) ⇒
        if (sample.length == windowSize) {
          val matchProbability = probabilityOfGestureEvent(sample)

          if (matchProbability >= threshold) {
            Some(Gesture(name, threshold))
          } else {
            Some(Neg(Gesture(name, threshold)))
          }
        } else {
          None
        }
      }
}
